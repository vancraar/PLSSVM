63template <kernel_type kernel, typename real_type, typename... Args>
64[[nodiscard]] inline real_type kernel_function(const std::vector<real_type> &xi, const std::vector<real_type> &xj, Args &&...args) {
65 using namespace plssvm::operators;
66
67 PLSSVM_ASSERT(xi.size() == xj.size(), "Sizes mismatch!: {} != {}", xi.size(), xj.size());
68
69 if constexpr (kernel == kernel_type::linear) {
70 static_assert(sizeof...(args) == 0, "Illegal number of additional parameters! Must be 0.");
71 return transposed{ xi } * xj;
72 } else if constexpr (kernel == kernel_type::polynomial) {
73 static_assert(sizeof...(args) == 3, "Illegal number of additional parameters! Must be 3.");
74 const auto degree = static_cast<real_type>(detail::get<0>(args...));
75 const auto gamma = static_cast<real_type>(detail::get<1>(args...));
76 const auto coef0 = static_cast<real_type>(detail::get<2>(args...));
77 return std::pow(std::fma(gamma, (transposed<real_type>{ xi } * xj), coef0), degree);
78 } else if constexpr (kernel == kernel_type::rbf) {
79 static_assert(sizeof...(args) == 1, "Illegal number of additional parameters! Must be 1.");
80 const auto gamma = static_cast<real_type>(detail::get<0>(args...));
81 return std::exp(-gamma * squared_euclidean_dist(xi, xj));
82 } else {
83 static_assert(detail::always_false_v<real_type>, "Unknown kernel type!");
84 }
85}
