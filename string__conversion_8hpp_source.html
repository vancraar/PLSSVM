39template <typename T, typename Exception = std::runtime_error>
40[[nodiscard]] inline T convert_to(std::string_view str) {
41 // select conversion function depending on the provided type
42 const auto convert_from_chars = [](const std::string_view sv, auto &val) {
43 if constexpr (std::is_floating_point_v<T>) {
44 // convert the string to a floating point value
45 return fast_float::from_chars(sv.data(), sv.data() + sv.size(), val);
46 } else if constexpr (std::is_integral_v<T>) {
47 // convert the string to an integral value
48 return std::from_chars(sv.data(), sv.data() + sv.size(), val);
49 } else {
50 // can't convert the string to a non-arithmetic type
51 static_assert(always_false_v<T>, "Can only convert arithmetic types!");
52 }
53 };
54
55 // remove leading whitespaces
56 str = trim_left(str);
57
58 // convert string to value fo type T
59 T val;
60 auto res = convert_from_chars(str, val);
61 if (res.ec != std::errc{}) {
62 throw Exception{ fmt::format("Can't convert '{}' to a value of type {}!", str, plssvm::detail::arithmetic_type_name<T>()) };
63 }
